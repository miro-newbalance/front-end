🖥️ Примеры объектов в JavaScript

1. Примеры известных объектов в JavaScript

// Встроенные объекты JavaScript
console.log(document.title); // Объект document
console.log(window.innerWidth); // Объект window
console.log(navigator.userAgent); // Объект navigator
console.log(Math.PI); // Объект Math
console.log(new Date()); // Объект Date

2. Пример описания пользовательского объекта

// Создание пользовательского объекта "Автомобиль"
let car = {
    brand: "Toyota",
    model: "Camry",
    year: 2022,
    color: "черный",
    engine: "2.5",
    start: function() {
        return "Двигатель запущен";
    },
    getInfo() {
        return `${this.brand} ${this.model}, ${this.year} года`;
    }
};

console.log(car);

3. Пример доступа к свойствам объекта

let person = {
    name: "Анна",
    age: 25,
    "favorite color": "синий",
    city: "Москва"
};

// Доступ через точку
console.log(person.name);
console.log(person.age);

// Доступ через квадратные скобки
console.log(person["city"]);
console.log(person["favorite color"]); // для свойств с пробелами

4. Команда delete (удаление свойства)

let user = {
    name: "Петр",
    age: 30,
    email: "petr@mail.com",
    phone: "+7-999-123-45-67"
};

console.log("До удаления:", user);

// Удаляем свойство phone
delete user.phone;
console.log("После удаления phone:", user);

// Удаляем свойство email
delete user.email;
console.log("После удаления email:", user);

5. Проверка существования свойства (in и hasOwnProperty)

let book = {
    title: "Война и мир",
    author: "Л. Толстой",
    year: 1869
};

// Оператор in
console.log("title" in book);      // true
console.log("pages" in book);      // false

// Метод hasOwnProperty
console.log(book.hasOwnProperty("author"));  // true
console.log(book.hasOwnProperty("price"));   // false

// Проверка с undefined
if (book.title !== undefined) {
    console.log("Свойство title существует");
}

6. Перебор свойств объекта (for...in)

let student = {
    firstName: "Мария",
    lastName: "Иванова",
    group: "ПИ-201",
    averageScore: 4.8,
    isBudget: true
};

console.log("Перебор свойств объекта:");
for (let key in student) {
    console.log(key + ": " + student[key]);
}

// Получение массива ключей и значений
console.log("Ключи:", Object.keys(student));
console.log("Значения:", Object.values(student));
console.log("Пары ключ-значение:", Object.entries(student));
